TO SEE RANKINGS 
---------------

SELECT r.PostID, SUM(r.Number) as 'RANK' FROM Rank r INNER JOIN Users u WHERE u.UserID = r.UserID GROUP BY r.PostID ORDER BY RANK ASC


TO SEE WHAT POSTS BELONG TO WHAT USERS
-------
SELECT p.PostID, u.USERID, p.Title FROM Post p INNER JOIN Users u ON u.UserID = p.UserID ORDER BY u.UserID ASC


TO SEE RANKING FOR SPECIFIC
---
SELECT p.PostID, p.Title, p.PostDate, c.Type, SUM(r.Number) as 'RANK' FROM Rank r INNER JOIN Post p INNER JOIN Users u INNER JOIN Category c ON u.UserID = p.UserID AND p.CategoryID = c.CategoryID AND p.postID = r.postID WHERE u.UserID = "1" GROUP BY p.PostID ORDER BY RANK DESC
